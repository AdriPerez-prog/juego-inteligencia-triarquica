    <script>

        // Variables globales

        let currentTab = 'analitica';

        let currentPages = {

            analitica: 1,

            creativa: 1,

            practica: 1

        };

        const totalPages = {

            analitica: 8,

            creativa: 6,

            practica: 6

        };

        

        // Cambiar pestañas

        document.querySelectorAll('.tab').forEach(tab => {

            tab.addEventListener('click', function() {

                const tabName = this.getAttribute('data-tab');

                

                // Actualizar pestaña activa

                document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));

                this.classList.add('active');

                

                // Ocultar todos los contenedores

                document.querySelectorAll('.contenedor-preguntas').forEach(c => {

                    c.classList.remove('active');

                });

                

                // Mostrar contenedor seleccionado

                document.getElementById(`contenedor-${tabName}`).classList.add('active');

                

                currentTab = tabName;

            });

        });

        

        // Navegación entre preguntas

        function showPage(tab, page) {

            // Ocultar todas las preguntas de esta pestaña

            const container = document.getElementById(`contenedor-${tab}`);

            container.querySelectorAll('.pregunta').forEach(p => {

                p.style.display = 'none';

            });

            

            // Mostrar la pregunta seleccionada

            container.querySelector(`.pregunta[data-index="${page}"]`).style.display = 'block';

            

            // Actualizar paginación

            container.querySelectorAll('.pagina').forEach(p => {

                p.classList.remove('active');

                if (parseInt(p.getAttribute('data-page')) === page) {

                    p.classList.add('active');

                }

            });

            

            // Actualizar botones de navegación

            document.getElementById(`prev-${tab}`).disabled = page === 1;

            document.getElementById(`next-${tab}`).disabled = page === totalPages[tab];

            

            currentPages[tab] = page;

        }

        

        // Inicializar páginas

        showPage('analitica', 1);

        showPage('creativa', 1);

        showPage('practica', 1);

        

        // Event listeners para botones de navegación

        document.getElementById('prev-analitica').addEventListener('click', () => {

            showPage('analitica', currentPages.analitica - 1);

        });

        

        document.getElementById('next-analitica').addEventListener('click', () => {

            showPage('analitica', currentPages.analitica + 1);

        });

        

        document.getElementById('prev-creativa').addEventListener('click', () => {

            showPage('creativa', currentPages.creativa - 1);

        });

        

        document.getElementById('next-creativa').addEventListener('click', () => {

            showPage('creativa', currentPages.creativa + 1);

        });

        

        document.getElementById('prev-practica').addEventListener('click', () => {

            showPage('practica', currentPages.practica - 1);

        });

        

        document.getElementById('next-practica').addEventListener('click', () => {

            showPage('practica', currentPages.practica + 1);

        });

        

        // Event listeners para paginación

        document.querySelectorAll('#contenedor-analitica .pagina').forEach(page => {

            page.addEventListener('click', function() {

                const pageNum = parseInt(this.getAttribute('data-page'));

                showPage('analitica', pageNum);

            });

        });

        

        document.querySelectorAll('#contenedor-creativa .pagina').forEach(page => {

            page.addEventListener('click', function() {

                const pageNum = parseInt(this.getAttribute('data-page'));

                showPage('creativa', pageNum);

            });

        });

        

        document.querySelectorAll('#contenedor-practica .pagina').forEach(page => {

            page.addEventListener('click', function() {

                const pageNum = parseInt(this.getAttribute('data-page'));

                showPage('practica', pageNum);

            });

        });

        

        // Evaluar respuestas

        document.getElementById('btn-evaluar').addEventListener('click', function() {

            // Validar que todas las secciones tengan al menos una respuesta

            let allAnswered = true;

            

            // Verificar respuestas analíticas

            for (let i = 1; i <= 8; i++) {

                if (!document.querySelector(`input[name="p${i}"]:checked`)) {

                    allAnswered = false;

                    alert(`Por favor responde la pregunta ${i} de la sección Analítica`);

                    showPage('analitica', Math.ceil(i / 1));

                    document.querySelector(`.tab-analitica`).click();

                    return;

                }

            }

            

            // Verificar respuestas creativas

            for (let i = 9; i <= 14; i++) {

                if (!document.getElementById(`p${i}`).value.trim()) {

                    allAnswered = false;

                    alert(`Por favor responde la pregunta ${i} de la sección Creativa`);

                    showPage('creativa', i - 8);

                    document.querySelector(`.tab-creativa`).click();

                    return;

                }

            }

            

            // Verificar respuestas prácticas

            for (let i = 15; i <= 20; i++) {

                if (!document.querySelector(`input[name="p${i}"]:checked`)) {

                    allAnswered = false;

                    alert(`Por favor responde la pregunta ${i} de la sección Práctica`);

                    showPage('practica', i - 14);

                    document.querySelector(`.tab-practica`).click();

                    return;

                }

            }

            

            if (allAnswered) {

                // Mostrar feedback

                document.getElementById('feedback').style.display = 'block';

                

                // Animación de barras de progreso (valores simulados)

                setTimeout(() => {

                    document.getElementById('progreso-analitica').style.width = '85%';

                    document.getElementById('progreso-creativa').style.width = '75%';

                    document.getElementById('progreso-practica').style.width = '90%';

                    

                    // Actualizar porcentajes

                    document.querySelectorAll('.resultado-item p')[0].textContent = '85%';

                    document.querySelectorAll('.resultado-item p')[2].textContent = '75%';

                    document.querySelectorAll('.resultado-item p')[4].textContent = '90%';

                }, 100);

                

                // Desplazarse al feedback

                document.getElementById('feedback').scrollIntoView({ behavior: 'smooth' });

                

                // Cambiar texto del botón

                this.textContent = '¡Volver a intentar!';

            }

        });
